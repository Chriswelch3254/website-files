(function bannerBoot(){
  if (document.readyState === "loading") { document.addEventListener("DOMContentLoaded", bannerBoot); return; }

  // Prevent double-binding if pasted twice
  if (window.__NF_BUMP_BANNER_BOOTED__) return;
  window.__NF_BUMP_BANNER_BOOTED__ = true;

  var banner   = document.getElementById("nf-bump-banner");
  if(!banner) return;

  // ✅ FIX: scope inner to this banner
  var inner    = banner.querySelector(".nf-bump-inner");
  var textEl   = document.getElementById("nf-bump-text");
  var closeBtn = document.getElementById("nf-bump-close");
  if(!inner || !textEl || !closeBtn) return;

  /* ===== Visibility / TTL / paths ===== */
  var STORAGE_KEY   = "nfBumpClosedAt";
  var HIDE_FOR_DAYS = 7;
  // Leave [] to show wherever you pasted it; otherwise list exact paths (lowercase, trailing slash optional).
  var ALLOWED_PATHS = [];
  var FORCE = (new URLSearchParams(location.search).get("nfdebug") === "1");

  function daysMs(d){ return d*24*60*60*1000; }
  function safeGet(k){ try{return localStorage.getItem(k);}catch(e){return null;} }
  function safeSet(k,v){ try{localStorage.setItem(k,v);}catch(e){} }
  function pathAllowed(){
    if (!ALLOWED_PATHS.length) return true;
    try{
      var p=(location.pathname||"/").toLowerCase().replace(/\/+$/,"");
      return ALLOWED_PATHS.indexOf(p)>-1 || ALLOWED_PATHS.indexOf(p+"/")>-1;
    }catch(e){ return true; }
  }
  function isHidden(){
    return banner.hasAttribute("hidden") || banner.style.display === "none";
  }
  function show(){ banner.removeAttribute("hidden"); banner.style.display="block"; }
  function hide(){ banner.setAttribute("hidden",""); banner.style.display="none"; }

  var closedAt = parseInt(safeGet(STORAGE_KEY)||"0",10);
  var ttlExpired = !closedAt || (Date.now()-closedAt) > daysMs(HIDE_FOR_DAYS);
  if (FORCE || (pathAllowed() && ttlExpired)) { show(); } else { hide(); return; }

  /* ===== Continuous single-line fit (no breakpoints/clamps) ===== */
  function fitOnce(){
    if (isHidden()) return;

    // Reset scaling before measuring
    textEl.style.transform = "scale(1)";
    textEl.style.maxWidth = "";

    var innerW = inner.clientWidth;
    var gap = 12;
    var closeW = closeBtn.getBoundingClientRect().width;

    // tiny breathing room to avoid subpixel wrap
    var avail = innerW - closeW - gap - 2;
    if (avail <= 0) return;

    textEl.style.maxWidth = avail + "px";

    var needsScale = textEl.scrollWidth > avail;
    if (needsScale){
      var scale = avail / textEl.scrollWidth;
      if (scale < 1){
        textEl.style.transformOrigin = "center left";
        textEl.style.transform = "scale(" + scale + ")";
      }
    }
  }

  // Debounced resize for smoothness
  var t;
  function onResize(){ clearTimeout(t); t=setTimeout(fitOnce, 40); }

  function unbind(){
    try{ window.removeEventListener("resize", onResize); }catch(e){}
  }

  closeBtn.addEventListener("click", function(){
    hide();
    safeSet(STORAGE_KEY, String(Date.now()));
    unbind();
    if (typeof window.gtag==="function") {
      try{ gtag("event","nf_info_banner_close",{page_path:location.pathname}); }catch(_){}
    }
  });

  // Initial fit and on resize/orientation
  requestAnimationFrame(fitOnce);
  window.addEventListener("resize", onResize);

  // If something else hides the banner, stop work
  document.addEventListener("visibilitychange", function(){
    if (document.visibilityState !== "visible") return;
    if (isHidden()) unbind();
    else fitOnce();
  });
})();
