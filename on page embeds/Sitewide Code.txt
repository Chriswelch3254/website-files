(function(){
  'use strict';

  /**
   * NF UTM Persistence v1.1 (CONSENT-GATED)
   * - Stores UTMs/click IDs from first touch
   * - Restores them into the URL via history.replaceState on later steps
   * - Prevents "file://C:/Users/..." and staging traffic from polluting production
   *
   * NOTE: Runs ONLY on /dlter* paths.
   */
  (function () {
    "use strict";

    var STORAGE_KEY = "nf.utm.persist.v1";
    var MAX_AGE_DAYS = 30;
    var MAX_AGE_MS = MAX_AGE_DAYS * 24 * 60 * 60 * 1000;

    var UTM_KEYS = ["utm_source", "utm_medium", "utm_campaign", "utm_term", "utm_content"];
    var CLICK_KEYS = ["gclid", "gbraid", "wbraid", "msclkid", "fbclid", "ttclid"];

    function hasConsent(){
      try {
        if (window.neuform && window.neuform.consent && typeof window.neuform.consent.isGranted === 'function') {
          return !!window.neuform.consent.isGranted();
        }
      } catch(e) {}
      try {
        var s = String(localStorage.getItem('nf.consent') || '').toLowerCase();
        if (s === 'granted') return true;
        if (s === 'denied') return false;
        var l = localStorage.getItem('nf.cookie.ok');
        return l === '1';
      } catch(e2) {
        return false;
      }
    }

    function isTrackableEnv() {
      try {
        if (location.protocol !== "https:" && location.protocol !== "http:") return false;
        var h = (location.hostname || "").toLowerCase();
        if (!h) return false;
        if (h === "localhost" || h === "127.0.0.1" || h === "[::1]") return false;
        if (h.endsWith("webflow.io")) return false; // staging
        return true;
      } catch (e) { return false; }
    }

    function isDlterPath() {
      try {
        return (location.pathname || "").toLowerCase().indexOf("/dlter") === 0;
      } catch (e) { return false; }
    }

    function parseQuery(qs) {
      var out = {};
      if (!qs) return out;
      qs.replace(/^\?/, "").split("&").forEach(function (part) {
        if (!part) return;
        var idx = part.indexOf("=");
        var k = idx > -1 ? part.slice(0, idx) : part;
        var v = idx > -1 ? part.slice(idx + 1) : "";
        try { k = decodeURIComponent(k || "").trim(); } catch (e) { k = (k || "").trim(); }
        try { v = decodeURIComponent(v || "").trim(); } catch (e2) { v = (v || "").trim(); }
        if (k) out[k] = v;
      });
      return out;
    }

    function pick(obj, keys) {
      var out = {};
      keys.forEach(function (k) {
        if (obj[k]) out[k] = obj[k];
      });
      return out;
    }

    function safeRead() {
      try {
        var raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return null;
        var data = JSON.parse(raw);
        if (!data || typeof data !== "object") return null;
        if (typeof data.ts !== "number") return null;
        if ((Date.now() - data.ts) > MAX_AGE_MS) return null;
        return data;
      } catch (e) { return null; }
    }

    function safeWrite(payload) {
      try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(payload));
      } catch (e) {}
    }

    function merge(a, b) {
      var out = {};
      Object.keys(a || {}).forEach(function (k) { out[k] = a[k]; });
      Object.keys(b || {}).forEach(function (k) { out[k] = b[k]; });
      return out;
    }

    function mergeMissing(base, add) {
      var out = {};
      Object.keys(base || {}).forEach(function (k) { out[k] = base[k]; });
      Object.keys(add || {}).forEach(function (k) {
        if (out[k] == null || out[k] === '') out[k] = add[k];
      });
      return out;
    }

    function buildQuery(obj) {
      var parts = [];
      Object.keys(obj || {}).forEach(function (k) {
        var v = obj[k];
        if (v == null || v === "") return;
        parts.push(encodeURIComponent(k) + "=" + encodeURIComponent(v));
      });
      return parts.join("&");
    }

    // Only run on real web env + only on DLTER URLs
    if (!isTrackableEnv()) return;
    if (!isDlterPath()) return;

    // ✅ Consent gate: no attribution persistence without consent
    if (!hasConsent()) {
      window.NF_NL = window.NF_NL || {};
      window.NF_NL.utm = {};
      return;
    }

    var q = parseQuery(location.search);
    var utm = pick(q, UTM_KEYS);
    var click = pick(q, CLICK_KEYS);
    var hasIncoming = Object.keys(utm).length > 0 || Object.keys(click).length > 0;

    var stored = safeRead();
    var storedParams = (stored && stored.params) ? stored.params : {};

    // Save/update if we have incoming attribution
    if (hasIncoming) {
      var combined = merge(storedParams, merge(utm, click));
      safeWrite({ ts: Date.now(), params: combined });
      storedParams = combined;
    }

    // Expose for other scripts (optional)
    window.NF_NL = window.NF_NL || {};
    window.NF_NL.utm = storedParams;

    // Restore UTMs into URL if missing on this step (helps Memberstack capture attribution)
    // - Never overwrites existing query params (e.g., session_id, ms_*)
    if (!hasIncoming && storedParams && Object.keys(storedParams).length > 0 && !q.utm_source) {
      var needsRestore = false;
      Object.keys(storedParams).forEach(function(k){
        if (!q[k]) needsRestore = true;
      });

      if (needsRestore) {
        var newParams = mergeMissing(q, storedParams);
        var newQs = buildQuery(newParams);
        if (newQs) {
          var curUrl = location.pathname + (location.search || "") + (location.hash || "");
          var newUrl = location.pathname + "?" + newQs + (location.hash || "");
          if (newUrl !== curUrl) history.replaceState(null, "", newUrl);
        }
      }
    }
  })();

  // NOTE:
  // - NF_NL.pantryIntel was moved OUT of Sitewide Code (global) to a Lab-only Pantry module/embed.
  // - This reduces page weight sitewide and prevents pantry API drift/duplication.

})();

function nfReviewsCacheKey(){

function nfReviewsCacheRead(ttlMs){
  try{
    var key = nfReviewsCacheKey();

    // TTL
    var ts = Number(obj.ts || 0);
    if (!ts || (Date.now() - ts) > (ttlMs || 0)) return null;

    if (!Array.isArray(obj.items)) return null;

    return obj.items;
  }catch(e){
    return null;
  }
}

/* --- Fetch + parse CSV (supports quoted commas) --- */
async function nfFetchReviews(opts){
  opts = opts || {};

  // If this page doesn't render reviews widgets, do nothing
  if (!opts.force && !nfPageNeedsReviews()) return null;

  // Dedupe concurrent calls
  if (window.neuform._reviewsPromise) return window.neuform._reviewsPromise;

  var TTL_MS = 6 * 60 * 60 * 1000; // 6 hours (session cache)

  // Serve from cache (fast path)
  if (!opts.forceRefresh){
    var cached = nfReviewsCacheRead(TTL_MS);
    if (cached){
      window.neuform.reviews = cached;
      document.dispatchEvent(new Event('nf:reviews-ready'));
      return cached;
    }
  }

  window.neuform._reviewsPromise = (async function(){
    try{
      // Optional: fetch timeout
      var controller = null, to = null;
      try{
        if ('AbortController' in window) {
          controller = new AbortController();
          to = setTimeout(function(){ try{ controller.abort(); }catch(_){ } }, 9000);
        }
      }catch(_){}

      // ✅ Removed cache:'no-store' (browser may cache; we also session-cache parsed output)
      var res = await fetch(window.neuform.REVIEWS_CSV, controller ? { signal: controller.signal } : undefined);
      if (to) { try{ clearTimeout(to); }catch(_){ } }

      var txt = await res.text();

      const rows = txt.trim()
        .split(/\r?\n/)
        .map(line => line.split(/,(?=(?:[^"]*"[^"]*")*[^"]*$)/)
          .map(c => c.replace(/^"|"$/g,'').replace(/""/g,'"')));

      if(!rows.length) throw new Error('Empty CSV');

      const headers = rows.shift().map(h => h.trim().toLowerCase());

      /* ✅ Privacy FAIL-CLOSED: never load a CSV that contains any email column */
      if (
        headers.includes('email') ||
        headers.includes('e-mail') ||
        headers.includes('e-mail address') ||
        headers.includes('email address')
      ){
        throw new Error('Privacy block: Reviews CSV contains an email column. Publish a public tab without it.');
      }

      const idx = Object.fromEntries(headers.map((h,i)=>[h,i]));
      const pick = (row, names, def='') => {
        for(const n of names){
          if(n in idx && row[idx[n]] !== undefined && String(row[idx[n]]).trim() !== '') {
            return row[idx[n]];
          }
        }
        return def;
      };

      const items = rows.map(r => {
        const quote = pick(r, ['quote','review','testimonial','comment'], '');

        // Metrics (combined + metric1..6 / result1..6)
        let metrics = [];
        const combined = pick(r, ['metrics','results'], '');
        if(combined){
          const split = combined
            .split(/\s*[·|]\s*|,(?![^(]*\))/)
            .map(s=>s.trim())
            .filter(Boolean);
          metrics = metrics.concat(split);
        }
        for(let i=1;i<=6;i++){
          const m = pick(r, [`metric${i}`, `result${i}`], '').trim();
          if(m) metrics.push(m);
        }
        metrics = metrics
          .map(m => m.replace(/\s+/g,' ').trim())
          .filter(m => m && !/^[\u2014\u2013\-–—\s.]+$/.test(m));

        const approvedRaw = String(pick(r, ['approved','publish','published'], '')).trim().toLowerCase();
        const permissionRaw = String(pick(r, ['permission','consent'], '')).trim().toLowerCase();

        const approved =
          (approvedRaw === '') ? false : (approvedRaw === 'true');  // explicit field if you use it
        const permission =
          (permissionRaw === 'true'); // must be true

        return {
          name:       pick(r, ['name','client','first name','full name'], ''),
          /* ✅ Removed: email field is never parsed/stored on-site */
          quote,
          rating:     parseFloat(pick(r, ['rating','stars','score'], '')),
          planName:   pick(r, ['planname','plan name','program','program name'], ''),
          planSlug:   String(pick(r, ['planslug','plan slug','slug'], '')).trim(),
          metrics:    metrics.join(' · '),
          date:       pick(r, ['date','review date','submitted'], ''),
          approved,
          permission
        };
      })
      .filter(x => {
        if(!x.permission) return false;
        if(!x.planSlug) return false;
        if(window.neuform.REVIEWS_REQUIRE_APPROVED) return !!x.approved;
        return true; // automatic mode
      });

      // Store sanitized items globally (no email possible)
      window.neuform.reviews = items;

      // Cache sanitized JSON for this tab/session
      nfReviewsCacheWrite(items);

      document.dispatchEvent(new Event('nf:reviews-ready'));
      return items;
    } catch (e){
      console.warn('Reviews fetch failed', e);
      window.neuform.reviews = [];
      try {
        (window.dataLayer = window.dataLayer || []).push({ event:'nf_reviews_error' });
      } catch(_){}
      document.dispatchEvent(new Event('nf:reviews-ready'));
      return [];
    } finally {
      window.neuform._reviewsPromise = null;
    }
  })();

  return window.neuform._reviewsPromise;
}

function nfUpsertProductAggregateRating(res, planSlug){
  try{
    const existing = document.getElementById('nf-product-jsonld');
    if (existing) {
      let json;
      try {
        json = JSON.parse(existing.textContent || '{}');
      } catch(parseErr){
        console.warn('nfUpsertProductAggregateRating: invalid existing JSON-LD, injecting fresh Product node', parseErr);
        json = null;
      }

      if (json) {
        let product = null;
        if (Array.isArray(json['@graph'])) {
          product = json['@graph'].find(n => n['@type'] === 'Product') || null;
        } else if (json['@type'] === 'Product') {
          product = json;
        }

        if (product) {
          product.aggregateRating = {
            '@type': 'AggregateRating',
            ratingValue: String(res.avg),
            reviewCount: String(res.count)
          };

          const reviews = (window.neuform.reviews || [])
            .filter(x => x.planSlug === planSlug && x.quote && Number.isFinite(x.rating) && x.rating > 0)
            .slice(0,2)
            .map(x => ({
              '@type': 'Review',
              'author': x.name ? {'@type':'Person','name': String(x.name)} : undefined,
              'reviewBody': String(x.quote).slice(0,500),
              'datePublished': x.date || undefined,
              'reviewRating': {
                '@type':'Rating',
                'ratingValue': String(x.rating),
                'bestRating': '5',
                'worstRating': '1'
              }
            }));

          if (reviews.length){
            if (Array.isArray(product.review)) {
              product.review = product.review.concat(reviews);
            } else if (product.review) {
              product.review = [product.review].concat(reviews);
            } else {
              product.review = reviews;
            }
          }

          existing.textContent = JSON.stringify(json);
          return true;
        }
      }
    }

    const meta = nfReadProductMeta();
    const productNode = {
      '@context':'https://schema.org',
      '@type':'Product',
      '@id': meta['@id'],
      name: meta.name,
      url: meta.url,
      image: meta.image || undefined,
      sku: meta.sku || undefined,
      brand: meta.brand,
      aggregateRating: {
        '@type':'AggregateRating',
        ratingValue: String(res.avg),
        reviewCount: String(res.count)
      }
    };

    const s = document.createElement('script');
    s.type = 'application/ld+json';
    s.textContent = JSON.stringify(productNode);
    document.body.appendChild(s);

  }
}

/* --- Renderers (stars + results list + JSON-LD upsert) --- */
function nfRenderOnPage(){
  document.querySelectorAll('[id^="rating-"]').forEach(badge => {
    const id   = badge.id;
    const slug = id.replace(/^rating-/, '');
    const res  = (slug === 'all' || slug === 'sitewide') ? nfAllRating() : nfPlanRating(slug);
    if(!res){ badge.style.display = 'none'; return; }

    const bg    = badge.querySelector('.nf-stars--bg');
    const fg    = badge.querySelector('.nf-stars--fg');
    const score = badge.querySelector('.nf-score');
    const count = badge.querySelector('.nf-count');

    if(bg && !bg.textContent.trim()) bg.textContent = '★★★★★';
    if(fg && !fg.textContent.trim()) fg.textContent = '★★★★★';

    const pct = Math.max(0, Math.min(100, (res.avg/5)*100));
    if(fg) fg.style.width = pct + '%';

    if(score) score.textContent = `${res.avg.toFixed(1)}/5`;
    if(count) count.textContent = `(${res.count} review${res.count>1?'s':''})`;

    badge.setAttribute(
      'aria-label',
      `${res.avg.toFixed(1)} out of 5 based on ${res.count} review${res.count>1?'s':''}`
    );

    if(slug !== 'all' && slug !== 'sitewide'){
      if(!window.neuform._ldInjected){
        nfUpsertProductAggregateRating(res, slug);
        window.neuform._ldInjected = true;
      }
    }
  });

  /* Results lists (unchanged) */
  const ALL = (window.neuform.reviews || []).slice();
  const esc = s => String(s||'').replace(/[&<>"']/g, c => ({
    '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'
  }[c]));
  const fmtDate = (d) => {
    if(!d) return '';
    const ms = Date.parse(d);
    if(Number.isNaN(ms)) return esc(d);
    return new Date(ms).toLocaleDateString(undefined,{year:'numeric',month:'short',day:'numeric'});
  };
  const starHTML = (r) => {
    if(!Number.isFinite(r) || r<=0) return '';
    const pct = Math.max(0, Math.min(100, (r/5)*100));
    return `
      <span class="nf-stars" aria-label="${r.toFixed(1)} out of 5">
        <span class="nf-stars--bg">★★★★★</span>
        <span class="nf-stars--fg" style="width:${pct}%">★★★★★</span>
      </span>`;
  };

  const renderBatch = (wrapper, list, from, size) => {
    const chunk = list.slice(from, from+size);
    const html = chunk.map(x => {
      const name = esc(x.name || 'Verified Client');
      const role = esc(x.planName || x.planSlug || '');
      const quoteRaw = (x.quote || '').replace(/\r?\n/g,' ').trim();
      const quote = esc(quoteRaw);
      const dateStr = fmtDate(x.date);
      const metrics = (x.metrics || '').split(/\s*·\s*/).filter(Boolean).map(m=>esc(m));

      return `
        <div class="tst_wrap">
          <div class="tst_card">
            <div class="tst_head">
              <div class="tst_name">${name}</div>
              <div class="tst_role">${role}</div>
            </div>

            <div class="tst_meta">
              ${starHTML(Number(x.rating))}
              ${dateStr ? `<span class="tst_date">${dateStr}</span>` : ''}
            </div>

            ${quote ? `<p class="tst_para">${quote}</p>` : ''}

            ${metrics.length ? `
              <div class="tst_metrics">
                ${metrics.map(m => `<div class="tst_metric">${m}</div>`).join('')}
              </div>
            ` : ''}

            <div class="tst_disclaimer">
              Results vary based on training history, nutrition, and adherence.
            </div>
          </div>
        </div>
      `;
    }).join('');

    wrapper.insertAdjacentHTML('beforeend', html);
    return chunk.length;
  };

  document.querySelectorAll('[data-results-list], #results-list').forEach(wrapper => {
    const filter = (wrapper.getAttribute('data-filter') || '').trim();

    const list = ALL
      .filter(x => !filter || x.planSlug === filter)
      .sort((a,b) => {
        const ad = Date.parse(a.date||''); const bd = Date.parse(b.date||'');
        if(!Number.isNaN(ad) && !Number.isNaN(bd)) return bd - ad;
        if(!Number.isNaN(ad) && Number.isNaN(bd))  return -1;
        if(Number.isNaN(ad) && !Number.isNaN(bd))  return 1;
        return (b.rating||0) - (a.rating||0);
      });

    wrapper.innerHTML = '';
    if(!list.length){
      wrapper.innerHTML = '<p class="nf-muted">No published results yet.</p>';
      return;
    }

    const PAGE = 6;
    wrapper._nfList = list;
    wrapper._nfRendered = 0;

    const first = renderBatch(wrapper, list, 0, PAGE);
    wrapper._nfRendered += first;

    const moreBtn = document.getElementById('results-more');
    if(moreBtn){
      const sync = ()=> moreBtn.style.display = (wrapper._nfRendered >= list.length) ? 'none' : 'block';
      sync();

      if (!moreBtn._nfHandlers) moreBtn._nfHandlers = new Map();
      if (!moreBtn._nfHandlers.has(wrapper)) {
        const handler = () => {
          const added = renderBatch(wrapper, wrapper._nfList, wrapper._nfRendered, PAGE);
          wrapper._nfRendered += added;
          sync();
        };
        moreBtn.addEventListener('click', handler);
        moreBtn._nfHandlers.set(wrapper, handler);
      }
    }
  });
}

/* =====================================================
   Automatic Plan — 1-hour Intake Access Expiry (SITEWIDE)
   v4.6:
   - Supports BOTH nf_ap_access_until (primary) + countdownDateTime (legacy)
   - Keeps keys in sync
   - Only removes plan if member currently has it
   - Adds backoff on removal failures (no 10s hammer loop)
   - Stops interval when no expiry exists or after successful removal
===================================================== */
(function(){
  "use strict";

  window.neuform = window.neuform || {};
  var nf = window.neuform;

  if (nf.__NF_AP_EXPIRY_BOOTED__) return;
  nf.__NF_AP_EXPIRY_BOOTED__ = true;

  var FREE_PLAN_ID =
    (nf.config && nf.config.memberstack && nf.config.memberstack.plans && nf.config.memberstack.plans.ap_intake_access_1h)
      ? nf.config.memberstack.plans.ap_intake_access_1h
      : "pln_automatic-plan-intake-access-1-hour--ic1a607rg";

  var KEY_PRIMARY =
    (nf.config && nf.config.storageKeys && nf.config.storageKeys.apExpiryPrimary)
      ? nf.config.storageKeys.apExpiryPrimary
      : "nf_ap_access_until";

  var KEY_LEGACY =
    (nf.config && nf.config.storageKeys && nf.config.storageKeys.apExpiryLegacy)
      ? nf.config.storageKeys.apExpiryLegacy
      : "countdownDateTime";

  var CHECK_MS = 10000;
  var timer = null;

  // Backoff (session-scoped) so we don't hammer removePlan on repeated failures
  var FAIL_KEY = "nf.ap.expiry.backoff.v1";
  var failCount = 0;
  var backoffUntil = 0;

  function ssGet(k){ try { return sessionStorage.getItem(k); } catch(e){ return null; } }
  function ssSet(k,v){ try { sessionStorage.setItem(k, v); } catch(e){} }
  function ssDel(k){ try { sessionStorage.removeItem(k); } catch(e){} }

  (function loadBackoff(){
    try{
      var raw = ssGet(FAIL_KEY);
      if (!raw) return;
      var o = JSON.parse(raw);
      if (!o || typeof o !== "object") return;
      failCount = Number(o.n || 0) || 0;
      backoffUntil = Number(o.until || 0) || 0;
    }catch(e){}
  })();

  function persistBackoff(){
    try{
      if (failCount <= 0 && !backoffUntil) { ssDel(FAIL_KEY); return; }
      ssSet(FAIL_KEY, JSON.stringify({ n: failCount, until: backoffUntil }));
    }catch(e){}
  }

  function clearBackoff(){
    failCount = 0;
    backoffUntil = 0;
    persistBackoff();
  }

  function bumpBackoff(){
    failCount = Math.min(10, (failCount || 0) + 1);
    var ms = Math.min(30 * 60 * 1000, 30000 * Math.pow(2, failCount - 1));
    backoffUntil = Date.now() + ms;
    persistBackoff();
    return ms;
  }

  function lsGet(k){ return (nf.storage && nf.storage.get) ? nf.storage.get(k) : (function(){ try{ return localStorage.getItem(k); }catch(e){ return null; } })(); }
  function lsSet(k,v){ if (nf.storage && nf.storage.set) nf.storage.set(k,v); else { try{ localStorage.setItem(k,v); }catch(e){} } }
  function lsDel(k){ if (nf.storage && nf.storage.del) nf.storage.del(k); else { try{ localStorage.removeItem(k); }catch(e){} } }

  function dlPush(obj){
    if (nf.dl && typeof nf.dl.push === "function") { nf.dl.push(obj); return; }
    try { (window.dataLayer = window.dataLayer || []).push(obj); } catch(e){}
  }

  function parseISODate(s){
    if(!s) return null;
    var d = new Date(String(s));
    return isNaN(d.getTime()) ? null : d;
  }

  function getExpiryISO(){
    var iso = lsGet(KEY_PRIMARY) || lsGet(KEY_LEGACY);
    if (iso) {
      if (lsGet(KEY_PRIMARY) !== iso) lsSet(KEY_PRIMARY, iso);
      if (lsGet(KEY_LEGACY)  !== iso) lsSet(KEY_LEGACY,  iso);
    }
    return iso || null;
  }

  function memberstackReady(){
    if (nf.memberstack && typeof nf.memberstack.ready === "function") return nf.memberstack.ready();
    var ms = window.$memberstackDom;
    return ms && typeof ms.removePlan === "function";
  }

  function stopInterval(){
    if (timer) {
      try { clearInterval(timer); } catch(_){ }
      timer = null;
    }
  }

  function ensureInterval(){
    if (timer) return;
    timer = setInterval(tick, CHECK_MS);
  }

  function memberHasPlan(member, planId){
    try{
      var d = member && member.data ? member.data : member;
      if (!d) return false;

      var pcs = d.planConnections;
      if (Array.isArray(pcs)) {
        for (var i=0;i<pcs.length;i++) {
          var pc = pcs[i];
          if (!pc) continue;
          if (pc.planId === planId) return true;
          if (pc.plan && pc.plan.id === planId) return true;
          if (pc.plan && pc.plan.planId === planId) return true;
        }
      }

      var plans = d.plans;
      if (Array.isArray(plans)) {
        for (var j=0;j<plans.length;j++) {
          var p = plans[j];
          if (!p) continue;
          if (p.id === planId || p.planId === planId) return true;
          if (p.plan && (p.plan.id === planId || p.plan.planId === planId)) return true;
        }
      }

      var planIds = d.planIds;
      if (Array.isArray(planIds)) {
        for (var k=0;k<planIds.length;k++) {
          if (String(planIds[k]) === planId) return true;
        }
      }
    }catch(e){}
    return false;
  }

  async function attemptRemoval(){
    if (backoffUntil && Date.now() < backoffUntil) return;

    var iso = getExpiryISO();

    // No expiry set: stop recurring checks
    if (!iso){
      clearBackoff();
      stopInterval();
      return;
    }

    var exp = parseISODate(iso);
    if (!exp){
      // Corrupt timestamp: clear keys and stop
      lsDel(KEY_PRIMARY);
      lsDel(KEY_LEGACY);
      clearBackoff();
      stopInterval();
      dlPush({ event: "nf_ap_access_expiry_invalid_cleared" });
      return;
    }

    // Not expired yet
    if (Date.now() <= exp.getTime()){
      clearBackoff();
      return;
    }

    // Memberstack not ready yet
    if (!memberstackReady()) return;

    // Validate plan id
    if (!FREE_PLAN_ID || typeof FREE_PLAN_ID !== "string" || FREE_PLAN_ID.indexOf("pln_") !== 0) {
      try { nf.util && nf.util.warn && nf.util.warn("[NF] Invalid AP plan id; cannot remove plan.", FREE_PLAN_ID); } catch(_){ }
      stopInterval();
      return;
    }

    var ms = (nf.memberstack && typeof nf.memberstack.get === "function") ? nf.memberstack.get() : window.$memberstackDom;
    if (!ms || typeof ms.removePlan !== "function") return;

    // Need a logged-in member to remove plan
    var mem = null;
    if (nf.memberstack && typeof nf.memberstack.getCurrentMemberSafe === "function") {
      mem = await nf.memberstack.getCurrentMemberSafe();
    } else if (typeof ms.getCurrentMember === "function") {
      try { mem = await ms.getCurrentMember(); } catch(e){ mem = null; }
    }

    if (!mem || !mem.data){
      // Not logged in (or no member payload yet) — keep checking
      return;
    }

    // If member no longer has the plan, expiry keys are stale; clear + stop
    if (!memberHasPlan(mem, FREE_PLAN_ID)) {
      lsDel(KEY_PRIMARY);
      lsDel(KEY_LEGACY);
      clearBackoff();
      stopInterval();
      dlPush({ event: "nf_ap_access_expired_plan_not_present" });
      return;
    }

    try {
      await ms.removePlan({ planId: FREE_PLAN_ID });

      lsDel(KEY_PRIMARY);
      lsDel(KEY_LEGACY);
      clearBackoff();

      dlPush({ event: "nf_ap_access_expired_removed" });

      stopInterval();

      try { location.reload(); } catch(_){ }
    } catch(e){
      var waitMs = bumpBackoff();
      console.warn("[NF] Intake access expiry removal failed (backing off " + Math.round(waitMs/1000) + "s):", e);
      dlPush({ event: "nf_ap_access_expired_remove_failed", nf_fail_count: failCount });
    }
  }

  function tick(){
    Promise.resolve(attemptRemoval()).catch(function(){});
  }

  // Run once immediately
  tick();

  // Start interval ONLY if an expiry exists right now
  if (getExpiryISO()) ensureInterval();

  // If tab becomes visible, re-check
  document.addEventListener("visibilitychange", function(){
    if (document.visibilityState === "visible") {
      tick();
      if (getExpiryISO()) ensureInterval();
    }
  });

  // If member logs in, re-check quickly
  try { document.addEventListener("memberstack:loggedIn", tick, { passive:true }); } catch(_){ }
  try { window.addEventListener("memberstack:loggedIn", tick, { passive:true }); } catch(_){ }

})();
</script>

    if (!gatedEl) return;

    // Only block elements you explicitly mark with data-nf-gated
    e.preventDefault();
    e.stopPropagation();

    var reason = '';
    try { reason = String(gatedEl.getAttribute('data-nf-gated') || '').trim(); } catch(_){ reason = ''; }
    var evt = (reason && reason.toLowerCase() === 'terms') ? 'nf_terms_block_global' : 'nf_gated_block';

    try {
      if (window.neuform && window.neuform.dl && typeof window.neuform.dl.push === "function") {
        window.neuform.dl.push({ event: evt, nf_reason: reason || null });
      } else {
        (window.dataLayer = window.dataLayer || []).push({ event: evt, nf_reason: reason || null });
      }
    } catch(_){}
  }, { capture:true, passive:false });
})();
