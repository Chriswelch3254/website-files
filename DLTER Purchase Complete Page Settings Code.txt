<!-- DLTER Purchase Router (v1.2 — JSON-safe + sessionStorage + safe redirect) -->
<script>
(function () {
  "use strict";

  var KEY = "nf.dlter.returnTo";
  var FALLBACK = "/dlter-quiz";
  var MAX_AGE_MS = 1000 * 60 * 60 * 6; // 6 hours

  function updateStatus(msg) {
    try {
      var el = document.getElementById("dlter-redirect-status");
      if (el) el.textContent = msg;
    } catch (e) {}
  }

  function readRawReturnTo() {
    var raw = null;

    // Try localStorage first
    try { raw = window.localStorage.getItem(KEY); } catch (e) {}

    // Fallback to sessionStorage (type page uses this fallback too)
    if (!raw) {
      try { raw = window.sessionStorage.getItem(KEY); } catch (e2) {}
    }

    // Clear both so it doesn't get "stuck"
    try { window.localStorage.removeItem(KEY); } catch (e3) {}
    try { window.sessionStorage.removeItem(KEY); } catch (e4) {}

    return raw;
  }

  function normalizeReturnTo(raw) {
    if (!raw) return null;
    raw = String(raw).trim();
    if (!raw) return null;

    // Back-compat: handle JSON payloads like {"path": "...", "ts": 123}
    if (raw.charAt(0) === "{") {
      try {
        var obj = JSON.parse(raw);
        if (obj && typeof obj.ts === "number" && (Date.now() - obj.ts) > MAX_AGE_MS) {
          return null; // expired
        }
        if (obj && typeof obj.path === "string") raw = obj.path;
      } catch (e) {
        // If JSON parse fails, fall through and treat as string
      }
    }

    raw = String(raw).trim();
    if (!raw) return null;

    // Prevent open redirects: only allow same-origin URLs/paths
    try {
      var u = new URL(raw, window.location.origin);
      if (u.origin !== window.location.origin) return null;
      return u.pathname + u.search + u.hash;
    } catch (e2) {
      // If URL parsing fails, treat as path
    }

    if (raw.charAt(0) !== "/") raw = "/" + raw;
    return raw;
  }

  function go(target) {
    try {
      window.location.replace(target);
    } catch (e) {
      window.location.href = target;
    }
  }

  // Small delay so Memberstack/GTM can finish any final work
  setTimeout(function () {
    try {
      var raw = readRawReturnTo();
      var target = normalizeReturnTo(raw) || FALLBACK;
      updateStatus("Redirecting…");
      go(target);
    } catch (err) {
      console.warn("[DLTER router] redirect failed, using fallback:", err);
      updateStatus("Redirecting…");
      go(FALLBACK);
    }
  }, 250);
})();
</script>
