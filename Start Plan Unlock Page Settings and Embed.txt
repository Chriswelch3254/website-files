<script>
  (function () {
    "use strict";

    // Idempotence guard (prevents double-run if embed is pasted twice)
    if (window.__NF_AP_UNLOCK_BOOTED__) return;
    window.__NF_AP_UNLOCK_BOOTED__ = true;

    // Start Plan Unlock (1 hour)
    // - UI-only timestamps (NOT enforcement)
    // - Best-effort storage only (never block redirect)
    // - Writes BOTH primary + legacy expiry keys for compatibility
    // - LocalStorage is the source of truth; sessionStorage is a best-effort mirror

    var WINDOW_MS = 60 * 60 * 1000;
    var untilISO = new Date(Date.now() + WINDOW_MS).toISOString();
    var nowISO = new Date().toISOString();

    function safe(fn){ try { fn(); } catch(_){} }

    // Prefer sitewide config if available; fall back safely
    var nf = window.neuform || (window.neuform = {});
    var keys = (nf.config && nf.config.storageKeys) ? nf.config.storageKeys : {};

    var KEY_PRIMARY = keys.apExpiryPrimary || "nf_ap_access_until";
    var KEY_LEGACY  = keys.apExpiryLegacy  || "countdownDateTime";
    var KEY_SET_AT  = "nf_ap_access_set_at";

    function tryDel(store, k){ try { store.removeItem(k); } catch(_){} }
    function trySet(store, k, v){ try { store.setItem(k, v); } catch(_){} }

    // Storage (best-effort)
    safe(function () {
      // Clear prior UI state (both stores)
      safe(function(){
        tryDel(localStorage, KEY_PRIMARY);
        tryDel(localStorage, KEY_LEGACY);
        tryDel(localStorage, KEY_SET_AT);
      });
      safe(function(){
        tryDel(sessionStorage, KEY_PRIMARY);
        tryDel(sessionStorage, KEY_LEGACY);
        tryDel(sessionStorage, KEY_SET_AT);
      });

      // Write unified expiry (localStorage = source of truth)
      safe(function(){
        trySet(localStorage, KEY_PRIMARY, untilISO);
        trySet(localStorage, KEY_LEGACY,  untilISO);
        trySet(localStorage, KEY_SET_AT,  nowISO);
      });

      // Mirror to sessionStorage (fallback / private mode compatibility)
      safe(function(){
        trySet(sessionStorage, KEY_PRIMARY, untilISO);
        trySet(sessionStorage, KEY_LEGACY,  untilISO);
        trySet(sessionStorage, KEY_SET_AT,  nowISO);
      });
    });

    // Optional analytics marker (no PII)
    safe(function () {
      window.neuform.track('nf_ap_unlock_redirect', {
        nf_access_window_mins: 60,
        nf_access_until: untilISO
      });
    });

    function redirect(){
      safe(function(){
        try { window.location.replace("/start-plan"); }
        catch(e){ window.location.href = "/start-plan"; }
      });
    }

    // Give UI a beat to paint so users aren’t confused post-purchase
    safe(function(){
      if (typeof requestAnimationFrame === "function") {
        requestAnimationFrame(function(){ setTimeout(redirect, 350); });
      } else {
        setTimeout(redirect, 350);
      }
    });

  })();
</script>
