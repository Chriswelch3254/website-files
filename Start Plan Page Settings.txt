<script>
(function () {
  "use strict";

  // Idempotence guard (prevents double-binding if pasted twice)
  if (window.__NF_START_PLAN_HASH_GUARD__) return;
  window.__NF_START_PLAN_HASH_GUARD__ = true;

  // Add stable page scope class
  try { document.body.classList.add('nf-start-plan-page'); } catch(e) {}

  // Keep the old class approach (useful if you want to scope anything else later)
  try { document.body.classList.add('nf-sticky-footer'); } catch(e) {}

  // Inject overlay element (dedicated layer; avoids body::before collisions)
  (function ensureOverlay(){
    try{
      if (document.getElementById('nf-start-plan-overlay')) return;
      var el = document.createElement('div');
      el.id = 'nf-start-plan-overlay';
      el.setAttribute('aria-hidden','true');

      var b = document.body;
      if (!b) return;

      // Insert as first child for predictable paint order
      if (b.firstChild) b.insertBefore(el, b.firstChild);
      else b.appendChild(el);
    }catch(e){}
  })();

  // Reduced motion flag (harmless + useful)
  try {
    if (window.matchMedia && window.matchMedia("(prefers-reduced-motion: reduce)").matches) {
      document.documentElement.classList.add("nf-reduce-motion");
    }
  } catch (e) {}

  // Prevent jump-to-top on Memberstack anchors (href="#") while preserving normal hash links.
  // IMPORTANT: Do NOT stop propagation. Memberstack relies on the click event to open modals / checkout.
  document.addEventListener("click", function (e) {
    var a = e.target && e.target.closest ? e.target.closest('a[href="#"]') : null;
    if (!a) return;

    // Only intercept for Memberstack-driven anchors
    if (
      a.hasAttribute("data-ms-price:add") ||
      a.hasAttribute("data-ms-modal") ||
      a.hasAttribute("data-ms-action")
    ) {
      e.preventDefault();
      // DO NOT: e.stopPropagation();
    }
  }, true);

})();
</script>
